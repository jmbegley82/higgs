struct higgs{
	char[MAX_ID_LENGTH]	identity;	// ex. "number1"
	char[4]			type;		// ex. "int", "char", "dub", "flot"
	uint16			count;		// if >0, this is an array
	void*			data;		// payload
}

create_higgs_int(char* identity, int* target) {
	higgs* root = malloc(sizeof higgs);
	root->identity = identity;
	root->count = 0;
	root->data = &target;
}
